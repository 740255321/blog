# 本地仓库

# 提交到本地仓库


git init //初始化本地仓库

git add 文件名 //(将文件提交到暂存区)

git commit -m "版本描述" //从暂存区提交本地仓库  注意：一定要加-m

git log //查看提交日志

# 移除暂存区文件


git reset HEAD 文件名

# 版本回退


git reset --hard HEAD^ //回退一个版本
git reset --hard HEAD^^ //回退两个版本

git reset --hard HEAD~ n //(n 代表回退几个版本数字)

# 版本前进

git reflog //查看所有版本信息标识字符串
git log --pretty==oneline //只查看版本标识和版本信息

git reset --hard 版本标识字符串 //（版本前进）

# 版本文件删除与恢复

// 删除
// 方法一
目录中删除后提交修改
git add git02.txt
git commit -m "删除"

// 方法二
git rm git02.txt

// 恢复
git checkout git02.txt
